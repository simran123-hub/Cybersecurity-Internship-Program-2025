import random

suits = ['Hearts', 'Diamonds', 'Spades', 'Clubs']
ranks = ['2', '3', '4', '5', '6', '7', '8', '9', '10', 'Jack', 'Queen', 'King', 'Ace']

deck = [rank + "of" + suit for suit in suits for rank in ranks]

random.shuffle(deck)

print("Shuffled Deck of Cards: ")
for cards in deck:
    print(cards)
